  graph: DS.belongsTo('graph', {async:true, inverse:'nodes'})
});

App.Edge = DS.Model.extend({
  weight: DS.attr('number'),
  nodes: DS.hasMany('node', {async:true, inverse:null}),

});

$e = [];
for($n = 1; $n <= $maxE || count($e) < $minE; $n++) {
  $e[] = [rand(1, $maxN), rand(1, $maxN)];
  // remove duplicates and self-loops
  $dup = [];
  foreach($e as $i => $v) {
    if ($v[0] == $v[1]) {
      unset($e[$i]);
      $n--;
    }
    $d = $v[0].':'.$v[1];
    if (isset($dup[$d])) {
      unset($e[$i]);
      $n--;
    } else {
      $dup[$d] = true;
    }
    foreach($e as $a => $j) {
      if ($j[0] == $v[1] && $j[1] == $v[0]) {
        unset($e[$a]);
      }
    }
  }
}
